#### To input elements of array

```cpp
#include <iostream>
using namespace std;
int main() {
    int x[5];
    for(int i = 0; i < 5; i++) {
        cout << "Enter array element " << i+1 << ": ";
        cin >> x[i];
    }
    for(int i = 0; i < 5; i++) {
        cout << x[i] << " ";
    }
    cout << endl;
}
```
